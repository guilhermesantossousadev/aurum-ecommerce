const finalizarCompra = async () => {
  const response = await fetch("https://seusite.com.br/api/pagamento/criar-preferencia", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      itens: anuncios.map((anuncio, index) => ({
        titulo: anuncio.titulo,
        quantidade: 1,
        preco: joias[index]?.valor ?? 0
      }))
    })
  });

  const data = await response.json();
  window.location.href = data.init_point;
};
